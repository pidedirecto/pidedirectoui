# DatePicker
Renders a date picker.

<Canvas of={DatePickerStories.Primary} />

The `onChange` prop will be called with the selected date

```tsx
import { DatePicker } from '@pidedirecto/ui'

const handleDate = (date: Date) => {
    // Handle date
}

<DatePicker
    value={value}
    label={'Select a date'}
    name='input'
    onChange={handleDate}
/>

```
