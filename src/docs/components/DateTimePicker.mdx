# DateTimePicker
Renders a date picker. In order to use this component you should wrap it with `MuiPickersUtilsProvider` component from `@material-ui/pickers`.

<Canvas of={DateTimePickerStories.Primary} sourceState='none' />

The `onChange` prop will be called with the selected date

```tsx
import { DateTimePicker } from '@pidedirecto/ui'

const handleDate = (date: Date) => {
    // Handle date
}

<DateTimePicker
    value={value}
    label={'Select a date'}
    name='input'
    onChange={handleDate}
/>

```
